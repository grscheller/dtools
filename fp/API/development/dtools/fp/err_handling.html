 39class MB[D]:
 40    """Maybe monad - class wrapping a potentially missing value.
 41
 42    - where `MB(value)` contains a possible value of type `~D`
 43    - `MB()` semantically represent a non-existent or missing value of type `~D`
 44    - immutable semantics, map & bind return new instances
 45      - can store any value of any type with one exception
 46        - if `~D` is `Sentinel`, storing `Sentinel(MB)` results in a MB()
 47      - warning: hashability invalidated if contained value is mutated
 48      - warning: hashed values invalidated if `put` or `pop` methods called
 49    - unsafe methods `get` and `pop`
 50      - could raise `ValueError` if MB is empty and alt values not given
 51    - stateful methods `put` and `pop`
 52      - useful to treat a `MB` as a stateful object
 53      - basically a container that can contain 1 or 0 objects
 54      - TODO: remove these, create a stateful object for this usecase
 55
 56    """

105    def get(self, alt: D | Sentinel = Sentinel('MB')) -> D | Never:
106        """Return the contained value if it exists, otherwise an alternate value.
107
108        - alternate value must be of type `~D`
109        - raises `ValueError` if an alternate value is not provided but needed
110
111        """
112        _sentinel: Final[Sentinel] = Sentinel('MB')
113        if self._value is not _sentinel:
114            return cast(D, self._value)
115        if alt is _sentinel:
116            msg = 'MB: an alternate return type not provided'
117            raise ValueError(msg)
118        return cast(D, alt)

120    def put(self, value: D) -> None:
121        """Put a value in the MB if empty, if not empty do nothing.
122
123        - warning: method will invalidate hashability if used
124
125        """
126        if self._value is Sentinel('MB'):
127            self._value = value
128
129    def pop(self) -> D | Never:
130        """Pop the value if the MB is not empty, otherwise fail.
131
132        - warning: method will invalidate hashability if used
133
134        """
135        _sentinel: Final[Sentinel] = Sentinel('MB')
136        if self._value is _sentinel:
137            msg = 'MB: Popping from an empty MB'
138            raise ValueError(msg)
139        popped = cast(D, self._value)
140        self._value = _sentinel
141        return popped

258class Xor[L, R]:
259    """Either monad - class semantically containing either a left or a right
260    value, but not both.
261
262    - implements a left biased Either Monad
263      - `Xor(value: ~L)` or `Xor(value: ~L, LEFT)` produces a left `Xor`
264      - `Xor(value: ~L, RIGHT)` produces a right `Xor`
265    - in a Boolean context
266      - `True` if a left `Xor`
267      - `False` if a right `Xor`
268    - two `Xor` objects compare as equal when
269      - both are left values or both are right values whose values
270        - are the same object
271        - compare as equal
272    - immutable, an `Xor` does not change after being created
273      - immutable semantics, map & bind return new instances
274        - warning: contained value need not be immutable
275        - warning: not hashable if value is mutable
276
277    """
278
279    __slots__ = '_value', '_side'
280    __match_args__ = ('_value', '_side')
281
282    T = TypeVar('T')
283    U = TypeVar('U')
284    V = TypeVar('V')
285    E = TypeVar('E')
286
287    @overload
288    def __new__(cls, value: L) -> Xor[L, R]: ...
289    @overload
290    def __new__(cls, value: L, side: Right) -> Xor[L, R]: ...
291    @overload
292    def __new__(cls, value: R, side: Left) -> Xor[L, R]: ...
293
294    def __new__(cls, value: L | R, side: Both = RIGHT) -> Xor[L, R]:
295        return super(Xor, cls).__new__(cls)
296
297    @overload
298    def __init__(self, value: L) -> None: ...
299    @overload
300    def __init__(self, value: L, side: Left) -> None: ...
301    @overload
302    def __init__(self, value: R, side: Right) -> None: ...
303
304    def __init__(self, value: L | R, side: Both = LEFT) -> None:
305        self._value = value
306        self._side = side
307
308    def __bool__(self) -> bool:
309        return self._side == LEFT
310
311    def __iter__(self) -> Iterator[L]:
312        if self:
313            yield cast(L, self._value)
314
315    def __repr__(self) -> str:
316        if self:
317            return 'Xor(' + repr(self._value) + ', LEFT)'
318        return 'Xor(' + repr(self._value) + ', RIGHT)'
319
320    def __str__(self) -> str:
321        if self:
322            return '< ' + str(self._value) + ' | >'
323        return '< | ' + str(self._value) + ' >'
324
325    def __len__(self) -> int:
326        # An Xor always contains just one value.
327        return 1
328
329    def __eq__(self, other: object) -> bool:
330        if not isinstance(other, type(self)):
331            return False
332
333        if self and other:
334            if (self._value is other._value) or (self._value == other._value):
335                return True
336
337        if not self and not other:
338            if (self._value is other._value) or (self._value == other._value):
339                return True
340
341        return False
342
343    def get(self) -> L | Never:
344        """Get value if a left. Unsafe convenience function.
345
346        - if the `Xor` is a "left" Xor
347          - return its value
348        - if the `Xor` contains a right value
349          - raises `ValueError`
350          - best practice is to first check the `Xor` in a boolean context
351
352        """
353        if self._side == RIGHT:
354            msg = 'Xor: get method called on a right valued Xor'
355            raise ValueError(msg)
356        return cast(L, self._value)
357
358    def get_left(self) -> MB[L]:
359        """Get value of `Xor` if a left. Safer version of `get` method.
360
361        - if `Xor` contains a left value, return it wrapped in a MB
362        - if `Xor` contains a tight value, return MB()
363
364        """
365        if self._side == LEFT:
366            return MB(cast(L, self._value))
367        return MB()
368
369    def get_right(self) -> MB[R]:
370        """Get value of `Xor` if a right
371
372        - if `Xor` contains a right value, return it wrapped in a MB
373        - if `Xor` contains a left value, return MB()
374
375        """
376        if self._side == RIGHT:
377            return MB(cast(R, self._value))
378        return MB()
379
380    def map[U](self, f: Callable[[L], U], right: R) -> Xor[U, R]:
381        """Map over if a left value.
382
383        - if `Xor` is a left then map `f` over its value
384          - if `f` successful return a left `Xor[U, R]`
385          - if `f` unsuccessful return right `Xor[S, R]`
386            - swallows any exceptions `f` may throw
387        - if `Xor` is a right
388          - return new `Xor(right=self._right): Xor[S, R]`
389          - use method `map_right` to adjust the returned value
390
391        """
392        if self._side == RIGHT:
393            return cast(Xor[U, R], self)
394
395        applied: MB[Xor[U, R]] = MB()
396        fallback: MB[Xor[U, R]] = MB()
397        try:
398            applied = MB(Xor(f(cast(L, self._value)), LEFT))
399        except (
400            LookupError,
401            ValueError,
402            TypeError,
403            BufferError,
404            ArithmeticError,
405            RecursionError,
406            ReferenceError,
407        ):
408            fallback = MB(cast(Xor[U, R], Xor(right, RIGHT)))
409
410        if fallback:
411            return fallback.get()
412        return applied.get()
413
414    def map_right(self, g: Callable[[R], R], alt_right: R) -> Xor[L, R]:
415        """Map over a right value."""
416        if self._side == LEFT:
417            return self
418
419        try:
420            right = g(cast(R, self._value))
421        except (
422            LookupError,
423            ValueError,
424            TypeError,
425            BufferError,
426            ArithmeticError,
427            RecursionError,
428            ReferenceError,
429        ):
430            right = alt_right
431
432        return Xor(right, RIGHT)
433
434    def change_right[V](self, right: V) -> Xor[L, V]:
435        """Change a right value's type and value."""
436        if self._side == LEFT:
437            return cast(Xor[L, V], self)
438        return Xor[L, V](right, RIGHT)
439
440    def bind[U](self, f: Callable[[L], Xor[U, R]], fallback_right: R) -> Xor[U, R]:
441        """Flatmap over the left value
442
443        - map over and then trivially "flatten" the left value
444        - propagate right values
445        - if bind fails, return alt_right wrapped in an right Xor
446          - WARNING: swallows RuntimeErrors
447
448        """
449        if self:
450            try:
451                return f(cast(L, self._value))
452            except (
453                LookupError,
454                ValueError,
455                TypeError,
456                BufferError,
457                ArithmeticError,
458                RecursionError,
459                ReferenceError,
460            ):
461                return Xor(fallback_right, RIGHT)
462        return cast(Xor[U, R], self)
463
464    @staticmethod
465    def call[U, V](f: Callable[[U], V], left: U) -> Xor[V, Exception]:
466        """Return Xor wrapped result of a function call that can fail"""
467        try:
468            xor = Xor[V, Exception](f(left), LEFT)
469        except (
470            LookupError,
471            ValueError,
472            TypeError,
473            BufferError,
474            ArithmeticError,
475            RecursionError,
476            ReferenceError,
477        ) as exc:
478            xor = Xor(exc, side=RIGHT)
479        return xor
480
481    @staticmethod
482    def lz_call[U, V](
483        f: Callable[[U], V], arg: U
484    ) -> Callable[[], Xor[V, Exception]]:
485        """Return an Xor of a delayed evaluation of a function"""
486
487        def ret() -> Xor[V, Exception]:
488            return Xor.call(f, arg)
489
490        return ret
491
492    @staticmethod
493    def idx[V](v: Sequence[V], ii: int) -> Xor[V, Exception]:
494        """Return an Xor of an indexed value that can fail"""
495        try:
496            xor = Xor[V, Exception](v[ii], LEFT)
497        except (
498            IndexError,
499            TypeError,
500            ArithmeticError,
501        ) as exc:
502            xor = Xor(exc, side=RIGHT)
503        return xor
504
505    @staticmethod
506    def lz_idx[V](v: Sequence[V], ii: int) -> Callable[[int], Xor[V, Exception]]:
507        """Return an Xor of a delayed indexing of a sequenced type that can fail"""
508
509        def ret(ii: int) -> Xor[V, Exception]:
510            return Xor.idx(v, ii)
511
512        return ret
513
514    @staticmethod
515    def sequence(itab_xor_lr: Iterable[Xor[L, R]], right: R) -> Xor[Iterator[L], R]:
516        """Sequence an indexable of type `Xor[~L, ~R]`
517
518        - if the iterated `Xor` values are all lefts, then
519          - return an `Xor` of an iterable of the left values
520        - otherwise return a right Xor containing the first right encountered
521
522        """
523        ts: list[L] = []
524
525        for xor_lr in itab_xor_lr:
526            if xor_lr:
527                ts.append(xor_lr.get())
528            else:
529                return Xor(right, RIGHT)
530
531        return Xor(iter(ts))
